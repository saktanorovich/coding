using System;
using System.Collections.Generic;

public class AvoidFour {
    const long MOD = 1000000007;

    long gcd(long a, long b)
    {
        while (b > 0)
        {
            long t = a % b;
            a = b;
            b = t;
        }
        return a;
    }

  public int count(long n) {
        long[,] m = new long[4, 4];
        m[0, 0] = 9;
        for (long i = 1; i < 4; ++i)
        {
            m[i, i - 1] = 1;
            m[0, i] = 9;
        }
        long[] start = new long[4];
        start[0] = 8;
        start[1] = 1;
        long[,] rm;
        long[,] dummy;
        sumpow(m, n, out rm, out dummy);
        if (n >= 44)
        {
            List<long> fours = new List<long>();
            fours.Add(44);
            while (fours[fours.Count - 1] * 10 + 4 <= n)
            {
                fours.Add(fours[fours.Count - 1] * 10 + 4);
            }
            for (int set = 1; set < (1 << fours.Count); ++set)
            {
                long lcm = 1;
                int am = 0;
                for (int i = 0; i < fours.Count; ++i)
                {
                    if ((set & (1 << i)) != 0)
                    {
                        ++am;
                        long z = gcd(lcm, fours[i]);
                        long by = fours[i] / z;
                        if (n / by < lcm)
                        {
                            lcm = n + 1;
                            break;
                        }
                        else
                        {
                            lcm *= by;
                        }
                    }
                }
                if (lcm > n)
                    continue;
                long[,] nrm;
                long[,] tmp = pow(m, lcm - 1);
                sumpow(mul(m, tmp), n / lcm, out nrm, out dummy);
                nrm = mul(tmp, nrm);
                if (am % 2 == 0)
                    rm = add(rm, nrm);
                else
                    rm = sub(rm, nrm);
            }
        }
        return (int)total(mul(rm, start));
  }

    void sumpow(long[,] m, long k, out long[,] sum, out long[,] max)
    {
        if (k == 0)
        {
            sum = new long[4, 4];
            max = unit();
        }
        else if (k % 2 == 0)
        {
            long[,] psum;
            long[,] pmax;
            sumpow(m, k / 2, out psum, out pmax);
            max = mul(pmax, pmax);
            sum = mul(psum, add(unit(), pmax));
        }
        else
        {
            long[,] psum;
            long[,] pmax;
            sumpow(m, k - 1, out psum, out pmax);
            max = mul(m, pmax);
            sum = add(unit(), mul(m, psum));
        }
    }

    private long[,] add(long[,] a, long[,] b)
    {
        long[,] c = new long[4, 4];
        for (int i = 0; i < 4; ++i)
            for (int j = 0; j < 4; ++j)
                c[i, j] = (c[i, j] + a[i, j] + b[i, j]) % MOD;
        return c;
    }

    private long[,] sub(long[,] a, long[,] b)
    {
        long[,] c = new long[4, 4];
        for (int i = 0; i < 4; ++i)
            for (int j = 0; j < 4; ++j)
                c[i, j] = (c[i, j] + a[i, j] - b[i, j] + MOD) % MOD;
        return c;
    }

    long[,] pow(long[,] m, long k)
    {
        if (k == 0)
            return unit();
        else if (k % 2 == 0)
            return pow(mul(m, m), k / 2);
        else
            return mul(m, pow(m, k - 1));
    }

    private long[,] mul(long[,] a, long[,] b)
    {
        long[,] c = new long[4, 4];
        for (int i = 0; i < 4; ++i)
            for (int j = 0; j < 4; ++j)
                for (int k = 0; k < 4; ++k)
                    c[i, j] = (c[i, j] + a[i, k] * b[k, j]) % MOD;
        return c;
    }

    private long total(long[] x)
    {
        long sum = 0;
        foreach (long a in x)
            sum = (sum + a) % MOD;
        return sum;
    }

    private long[] mul(long[,] a, long[] b)
    {
        long[] c = new long[4];
        for (int i = 0; i < 4; ++i)
            for (int k = 0; k < 4; ++k)
                c[i] = (c[i] + a[i, k] * b[k]) % MOD;
        return c;
    }

    private long[,] unit()
    {
        long[,] res = new long[4, 4];
        for (long i = 0; i < 4; ++i)
            res[i, i] = 1;
        return res;
    }
