import java.util.*;
public class Roxor {
  int[] f = new int[16];

  public boolean canwin(int[] piles) {
    int l = piles.length;
    int s = 0;
    for (int i=0; i<l; i++) {
      if (piles[i] % 2 == 1) {
        s = s ^ f[l-i-1];
      }
    }
    if (s != 0) {
      return true;
    } else {
      return false;
    }
  }

    public int[] play(int[] piles) {
  f[0] = 0;
  for (int i=1; i<=15; i++) {
    boolean[] used = new boolean[100];
    Arrays.fill(used, false);
    for (int j=0; j<i; j++) {
      for (int k=0; k<i; k++) {
        used[f[j]^f[k]] = true;
      }
    }
    f[i] = 0;
    while (used[f[i]]) {
      f[i]++;
    }
  }
  if (!canwin(piles)) {
    int[] res = new int[0];
    return res;
  } else {
    for (int i=0; i+1<piles.length; i++) {
      if (piles[i] > 0) {
        for (int j=i+1; j<piles.length; j++) {
          for (int k=j; k<piles.length; k++) {
            piles[i]--; piles[j]++; piles[k]++;
            if (!canwin(piles)) {
              int[] res = new int[3];
              res[0] = i; res[1] = j; res[2] = k;
              return res;
            }
            piles[i]++; piles[j]--; piles[k]--;
          }
        }  
      }
    }
  }
  int[] res = new int[0];
  return res;
   }

}
